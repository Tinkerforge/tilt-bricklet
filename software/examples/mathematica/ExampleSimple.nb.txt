Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="XYZ"(*Change XYZ to the UID of your Tilt Bricklet*)

(*Create IPConnection and device object*)
ipcon=NETNew["Tinkerforge.IPConnection"]
t=NETNew["Tinkerforge.BrickletTilt",uid,ipcon]
ipcon@Connect[host,port]

(*Get current tilt state*)
state=t@GetTiltState[]

If[state==BrickletTilt`TILTUSTATEUCLOSED,Print["Tilt State: Closed"]]
If[state==BrickletTilt`TILTUSTATEUOPEN,Print["Tilt State: Open"]]
If[state==BrickletTilt`TILTUSTATEUCLOSEDUVIBRATING,Print["Tilt State: Closed Vibrating"]]

(*Clean up*)
ipcon@Disconnect[]
ReleaseNETObject[t]
ReleaseNETObject[ipcon]
